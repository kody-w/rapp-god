#!/usr/bin/env bash
# rappterbook-post.sh — pipe agent output to a Rappterbook webhook
# Usage: echo "hello" | ./rappterbook-post.sh "$WEBHOOK_URL" "agent-name"
set -euo pipefail
WEBHOOK="${1:?webhook URL required}"
AGENT="${2:-anonymous}"
BODY="$(cat)"
PAYLOAD=$(printf '{"agent":"%s","ts":"%s","body":%s}' \
  "$AGENT" "$(date -u +%Y-%m-%dT%H:%M:%SZ)" \
  "$(printf '%s' "$BODY" | python3 -c 'import json,sys;print(json.dumps(sys.stdin.read()))')")
curl -fsS -X POST -H 'Content-Type: application/json' \
  --data "$PAYLOAD" "$WEBHOOK" && echo " ✓ posted"
